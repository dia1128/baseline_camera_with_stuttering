String getButtonTitle(Item item) {
  int count = 0;
  String title;
  if (state.storageRepo.getAction(item.name) != null) {
    count = state.storageRepo.getAction(item.name).count;
    title = '${item.name} ($count)';
  } else {
    title = '${item.name}';
  }
  return title;
}

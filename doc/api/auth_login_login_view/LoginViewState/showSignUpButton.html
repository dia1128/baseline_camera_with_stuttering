Widget showSignUpButton(BuildContext context) {
  return SafeArea(
    child: TextButton(
      child: Text('Don\'t have an account? Sign up.'),
      // tell auth navigator to show signup page
      onPressed: () => context.read<AuthCubit>().showSignUp(),
    ),
  );
}

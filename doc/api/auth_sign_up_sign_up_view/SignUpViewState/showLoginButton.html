Widget showLoginButton(BuildContext context) {
  return SafeArea(
    child: TextButton(
      child: Text('Already have an account? Sign in.'),
      // tell auth navigator to show login page
      onPressed: () => context.read<AuthCubit>().showLogin(),
    ),
  );
}
